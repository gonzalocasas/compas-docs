[docs]def clear_layers(layers):
    """ Deletes objects in given layers.

    Parameters:
        layers (list, str): Layers or 'all'.

    Returns:
        None
    """
    if layers == 'all':
        delete_all_objects()
    elif isinstance(layers, list):
        for layer in layers:
            delete_objects(get_objects(layer=layer))
